import { useState, useRef } from "react";

const ScrambleText = ({ text }) => {
  const [displayText, setDisplayText] = useState(text);
  const intervalRef = useRef(null);

  const getShuffled = () => {
    return [...text].map(char => (Math.random() < 0.5 ? char : String.fromCharCode(33 + Math.floor(Math.random() * 94)))).join("");
  };

  const handleMouseEnter = () => {
    if (intervalRef.current) return;

    // Shuffle letters every 50ms for smoother animation
    intervalRef.current = setInterval(() => {
      setDisplayText(getShuffled());
    }, 30);

    // After 0.75s, progressively restore original text
    setTimeout(() => {
      clearInterval(intervalRef.current);
      intervalRef.current = null;

      const original = text.split("");
      let step = 0;

      const unscrambleInterval = setInterval(() => {
        step++;
        const newDisplay = original.map((char, idx) => (idx < step ? char : String.fromCharCode(33 + Math.floor(Math.random() * 94)))).join("");
        setDisplayText(newDisplay);

        if (step >= original.length) clearInterval(unscrambleInterval);
      }, 30); // adjust speed of progressive unscramble
    }, 300);
  };

  return (
    <span
      className="cursor-pointer inline-block text-[#eeeeee]"
      onMouseEnter={handleMouseEnter}
    >
      {displayText}
    </span>
  );
};

export default ScrambleText;
